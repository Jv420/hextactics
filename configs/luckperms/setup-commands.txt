# HexTactics LuckPerms setup
# Plak deze commands in de console.

lp creategroup speler
lp creategroup vip
lp creategroup elite
lp creategroup legend
lp creategroup tactician
lp creategroup helper
lp creategroup moderator
lp creategroup admin
lp creategroup owner

lp setdefaultgroup speler

# Prefixes
lp group speler meta setprefix 10 "&7[Speler] "
lp group vip meta setprefix 20 "&a[VIP] "
lp group elite meta setprefix 30 "&b[Elite] "
lp group legend meta setprefix 40 "&6[Legend] "
lp group tactician meta setprefix 50 "&5[Tactician] "
lp group helper meta setprefix 60 "&e[Helper] "
lp group moderator meta setprefix 70 "&9[Mod] "
lp group admin meta setprefix 80 "&c[Admin] "
lp group owner meta setprefix 100 "&4[Owner] "

# Inheritance
lp group vip parent add speler
lp group elite parent add vip
lp group legend parent add elite
lp group tactician parent add legend
lp group helper parent add speler
lp group moderator parent add helper
lp group admin parent add moderator
